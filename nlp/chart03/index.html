        ÂèÇÊï∞:

        """
        super().__init__()
        self.d_model = d_model

        # ------------------------------
        # 1Ô∏è‚É£ ÊâãÂä®ÂÆö‰πâ Q„ÄÅK„ÄÅV ÁöÑÊùÉÈáçÁü©Èòµ (‰∏ç‰ΩøÁî® nn.Linear)

        self.W_K = nn.Parameter(torch.randn(d_model, d_model))
        self.W_V = nn.Parameter(torch.randn(d_model, d_model))

        # 2Ô∏è‚É£ ËæìÂá∫Â±ÇÊùÉÈáç (ÊääÊ≥®ÊÑèÂäõËæìÂá∫Êò†Â∞ÑÂõû d_model Áª¥Â∫¶)
        self.W_O = nn.Parameter(torch.randn(d_model, d_model))

        # 3Ô∏è‚É£ ÂèØÈÄâÂÅèÁΩÆ
        self.b_Q = nn.Parameter(torch.zeros(d_model))
        self.b_K = nn.Parameter(torch.zeros(d_model))
        self.b_V = nn.Parameter(torch.zeros(d_model))
        self.b_O = nn.Parameter(torch.zeros(d_model))

        # ÂàùÂßãÂåñÔºàÊ®°‰ªø nn.Linear ÁöÑ Xavier ÂàùÂßãÂåñÔºâ
        nn.init.xavier_uniform_(self.W_Q)
        nn.init.xavier_uniform_(self.W_K)
        nn.init.xavier_uniform_(self.W_V)
        nn.init.xavier_uniform_(self.W_O)

    def forward(self, x, mask=None):
        """
        ÂèÇÊï∞:
            x: (batch, seq_len, d_model)
            mask: (batch, seq_len, seq_len)ÔºåÂèØÈÄâ
        ËøîÂõû:

            attn_weights: (batch, seq_len, seq_len)

        d_k = d_model  # ÂçïÂ§¥Êó∂Ôºåd_k = d_model

        # ------------------------------
        # 4Ô∏è‚É£ Á∫øÊÄßÂèòÊç¢: XW + b
        #    Ê≥®ÊÑèÔºöx ÁöÑÂΩ¢Áä∂ (batch, seq_len, d_model)
        #    ÊâÄ‰ª•Ë¶ÅÂú®ÊúÄÂêé‰∏Ä‰∏™Áª¥Â∫¶‰∏äÂÅöÁü©Èòµ‰πòÊ≥ï
        # ------------------------------
        Q = torch.matmul(x, self.W_Q) + self.b_Q     # (batch, seq_len, d_model)
        K = torch.matmul(x, self.W_K) + self.b_K     # (batch, seq_len, d_model)
        V = torch.matmul(x, self.W_V) + self.b_V     # (batch, seq_len, d_model)

        # ------------------------------
        # 5Ô∏è‚É£ ËÆ°ÁÆóÊ≥®ÊÑèÂäõÂæóÂàÜ: QK^T / sqrt(d_k)
        # ------------------------------
        # K^T ÈúÄË¶ÅËΩ¨ÁΩÆÊúÄÂêé‰∏§‰∏™Áª¥Â∫¶ (seq_len, d_model) -> (d_model, seq_len)
        scores = torch.matmul(Q, K.transpose(-2, -1)) / (d_k ** 0.5)  # (batch, seq_len, seq_len)

        # ------------------------------
        # 6Ô∏è‚É£ mask (Â¶ÇÊûúÊúâ)
        # ------------------------------

        # ------------------------------

        # ------------------------------
        attn_weights = F.softmax(scores, dim=-1)  # (batch, seq_len, seq_len)

        # ------------------------------
        # 8Ô∏è‚É£ Âä†ÊùÉÊ±ÇÂíåÂæóÂà∞ËæìÂá∫: Attention(Q,K,V) = softmax(QK^T)V
        # ------------------------------
        out = torch.matmul(attn_weights, V)  # (batch, seq_len, d_model)

        # ------------------------------
        # 9Ô∏è‚É£ ËæìÂá∫Á∫øÊÄßÂ±Ç (ÊâãÂä®ÂÆûÁé∞)
        # ------------------------------
        out = torch.matmul(out, self.W_O) + self.b_O  # (batch, seq_len, d_model)

        return out, attn_weights

class MultiHeadSelfAttention(nn.Module):
    def __init__(self, d_model, num_heads):
        """
        Â§öÂ§¥Ëá™Ê≥®ÊÑèÂäõÊú∫Âà∂Ôºà‰∏ç‰ΩøÁî® nn.LinearÔºâ
        ÂèÇÊï∞:

            num_heads: Ê≥®ÊÑèÂäõÂ§¥ÁöÑÊï∞Èáè
        """
        super().__init__()
        assert d_model % num_heads == 0, "d_model ÂøÖÈ°ªËÉΩË¢´ num_heads Êï¥Èô§"

        self.d_model = d_model
        self.num_heads = num_heads
        self.d_k = d_model // num_heads  # ÊØè‰∏™Â§¥ÁöÑÁª¥Â∫¶

        # ---- ÂèØËÆ≠ÁªÉÂèÇÊï∞ ----
        # ÊØè‰∏™ Q/K/V ÊäïÂΩ±ÈÉΩÊúâËá™Â∑±ÁöÑ‰∏ÄÁªÑÊùÉÈáçÔºàÂÖ±‰∫´Âú®ÊâÄÊúâÂ§¥Ôºâ
        self.W_Q = nn.Parameter(torch.randn(d_model, d_model))
        self.W_K = nn.Parameter(torch.randn(d_model, d_model))
        self.W_V = nn.Parameter(torch.randn(d_model, d_model))

        # ËæìÂá∫Êò†Â∞ÑÊùÉÈáçÔºàÊãºÊé•ÂêéÁöÑÁ∫øÊÄßÂèòÊç¢Ôºâ
        self.W_O = nn.Parameter(torch.randn(d_model, d_model))

        # ÂÅèÁΩÆÈ°π
        self.b_Q = nn.Parameter(torch.zeros(d_model))
        self.b_K = nn.Parameter(torch.zeros(d_model))
        self.b_V = nn.Parameter(torch.zeros(d_model))
        self.b_O = nn.Parameter(torch.zeros(d_model))

        # ÂàùÂßãÂåñ
        nn.init.xavier_uniform_(self.W_Q)
        nn.init.xavier_uniform_(self.W_K)
        nn.init.xavier_uniform_(self.W_V)
        nn.init.xavier_uniform_(self.W_O)

    def forward(self, x, mask=None):
        """
        ÂèÇÊï∞:
            x: (batch, seq_len, d_model)
            mask: (batch, 1, seq_len, seq_len)ÔºåÂèØÈÄâ
        ËøîÂõû:

        batch_size, seq_len, _ = x.size()

        # 1Ô∏è‚É£ ÁîüÊàê Q, K, V
        Q = torch.matmul(x, self.W_Q) + self.b_Q   # (batch, seq_len, d_model)
        K = torch.matmul(x, self.W_K) + self.b_K
        V = torch.matmul(x, self.W_V) + self.b_V

        # 2Ô∏è‚É£ ÊãÜÂàÜÂ§öÂ§¥
        # Áª¥Â∫¶ÊãÜÂàÜÂêéÂΩ¢Áä∂: (batch, num_heads, seq_len, d_k)
        Q = Q.view(batch_size, seq_len, self.num_heads, self.d_k).transpose(1, 2)
        K = K.view(batch_size, seq_len, self.num_heads, self.d_k).transpose(1, 2)
        V = V.view(batch_size, seq_len, self.num_heads, self.d_k).transpose(1, 2)

        # 3Ô∏è‚É£ ËÆ°ÁÆóÊØè‰∏™Â§¥ÁöÑÊ≥®ÊÑèÂäõÂàÜÊï∞
        scores = torch.matmul(Q, K.transpose(-2, -1)) / (self.d_k ** 0.5)  # (batch, heads, seq_len, seq_len)

        # 4Ô∏è‚É£ Â∫îÁî®Êé©Á†ÅÔºàÂ¶ÇÊûúÊúâÔºâ
        if mask is not None:
            # mask Â∫îËØ•ÂèØÂπøÊí≠Âà∞ (batch, heads, seq_len, seq_len)
            scores = scores.masked_fill(mask == 0, float('-inf'))

        # 5Ô∏è‚É£ softmax ÂæóÂà∞Ê≥®ÊÑèÂäõÊùÉÈáç
        attn_weights = F.softmax(scores, dim=-1)

        # 6Ô∏è‚É£ ÂØπ V Âä†ÊùÉÊ±ÇÂíå

        # 8Ô∏è‚É£ ÊúÄÂêéÁöÑËæìÂá∫ÊäïÂΩ±
        out = torch.matmul(head_outputs, self.W_O) + self.b_O  # (batch, seq_len, d_model)

        return out, attn_weights
